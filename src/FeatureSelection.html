 73        """
 74        Register the stations in order to have a visualisation
 75
 76        Args:
 77            stations_dataframe (DataFrame | GeoDataFrame) : DataFrame containing the stations
 78            id_column (str) : column name of the stations_dataframe which contains the unique id of the station
 79            get_id_from_sensor_regex (str) : regex used to find the station id in the dataframe containing all records used for feature selection
 80            lon_column (str) : column name of the stations_dataframe which contains the longitude of the station
 81            lat_column (str) : column name of the stations_dataframe which contains the latitude of the station
 82            geometry_column (str | None) : if provided and dataframe is a GeoDataFrame, column name of the stations_dataframe which contains the geometry
 83            name_column (str | None) : if provided, column name of the stations_dataframe which contains the name of the station
 84            crs (str) : crs used for the location

 87        self._stations_dataframe = stations_dataframe
 88        self._stations_name_column = name_column
 89        self._stations_id_column = id_column
 90        self._stations_get_id_from_sensor_regex = get_id_from_sensor_regex
 91        self._stations_crs = crs

 93        if geometry_column:
 94            self._stations_geometry_column = geometry_column
 95        else:
 96            self._stations_dataframe.loc[:, "geometry"] = geopandas.points_from_xy(

140    def select(self, dataframe, target_columns, method_names=None):
141        """
142        Apply feature selection methods on target_columns for a given dataframe
143
144        Args:
145            dataframe (DataFrame) : dataframe which contains the data used to apply the feature selection. 1 column by feature and 1 line by entry
146            target_columns (str[]) : array of the target column names used to apply the feature selection
147            method_names (str[] | None) : array of the method names to use for feature selection, if None, all registered methods will be applied
148        """
149        methods = (
150            self._feature_selection_method_objects
151            if not method_names
152            else [
153                method
154                for method in self._feature_selection_method_objects
155                if method.get_method_name() in method_names
156            ]
157        )
158
159        for method in methods:
160            method.select(dataframe, target_columns)
161
162        self._last_used_methods = [method.get_method_name() for method in methods]
163        self._last_used_targets = target_columns

205        """
206        Plot the results of the feature selection. Feature selection (`select()`) must be done before.

214        if not used_targets:
215            used_targets = self._last_used_targets
216        if not used_methods:
217            used_methods = self._last_used_methods
218        if len(used_methods) > 1 and len(used_targets) > 1:

220                "Cannot plot results for multiple methods and multiple targets at once yet..."

223        if len(used_methods) > 1:
224            fig, axs = plt.subplots(

230            fig.suptitle(

234            for i, method in enumerate(used_methods):
235                self._plot(
236                    used_targets[0],
237                    method,
238                    axs[i, 0],
239                    axs[i, 1],
240                    title=f"Stations importance for the method {method}",
241                )
242        elif len(used_targets) > 1:
243            fig, axs = plt.subplots(
244                len(used_targets),
245                2,
246                figsize=(30, 10 * len(used_targets)),
247                gridspec_kw={"width_ratios": [3, 1]},
248            )
249            fig.suptitle(
250                f"Feature importance visualization for the method {used_methods[0]}",
251                fontsize=36,
252            )
253            for i, target in enumerate(used_targets):
254                self._plot(
255                    target,
256                    used_methods[0],
257                    axs[i, 0],
258                    axs[i, 1],
259                    title=f"Stations importance for the target {target}",
260                )
261        else:
262            fig, (ax1, ax2) = plt.subplots(
263                1, 2, figsize=(30, 10), gridspec_kw={"width_ratios": [3, 1]}
264            )
265            fig.suptitle(
266                f"Feature importance visualization for the method {used_methods[0]} and the target {used_targets[0]}",
267                fontsize=36,
268            )
269            self._plot(used_targets[0], used_methods[0], ax1, ax2)

271    def _plot(self, target, method, ax1, ax2, title="Stations importance"):
272        """
273        Private method. Plot the result for a target and a method. Feature selection (`select()`) must be done before
274
275        Args:
276            target (str) : target name (must be referenced in `target_columns` when `select()`)
277            method (str) : method name (must be referenced in `method_names` when `select()`, or None used)
278            ax1 (Axe) : axe which will contains the map
279            ax2 (Axe) : axe which will contains the heatmap
280            title (str) : title of the figure
281        """
282
283        stations_importance = self.get_stations_importance(target, method)
284        stations_importance = stations_importance.to_crs(
285            epsg=3857
286        )  # change to Spherical Mercator to add ctx base map properly
287        features_importance = self.get_features_importance()
288        stations_importance.plot(
289            ax=ax1,
290            column="max_importance_value",
291            legend=True,
292            markersize=(stations_importance["nb_important_sensors"] * 40 + 5),
293            cmap=plt.cm.get_cmap("plasma"),
294            vmin=0,
295            vmax=1,
296        )
297        ax1.set_xlabel("Longitude", fontsize=10)
298        ax1.set_ylabel("Latitude", fontsize="medium")
299        ax1.set_title(title)
300        ctx.add_basemap(ax1, source=ctx.providers.CartoDB.Positron)
301        for x, y, label, offsetY in zip(
302            stations_importance.geometry.x,
303            stations_importance.geometry.y,
304            stations_importance[self._stations_id_column],
305            stations_importance["nb_important_sensors"] + 5,
306        ):
307            ax1.annotate(
308                label, xy=(x, y), xytext=(0, offsetY), textcoords="offset points"
309            )
310
311        features_importance = (
312            features_importance[method]
313            .dropna()
314            .sort_values(by=[target], ascending=False)
315        )
316        sns.heatmap(
317            features_importance[[target]],
318            ax=ax2,
319            annot=True,
320            linewidths=0.5,
321            cmap=plt.cm.get_cmap("plasma"),
322            cbar=False,
323            vmin=0,
324            vmax=1,
325        )

327    def get_features_importance(self):
328        """
329        Get the features importance. Feature selection (`select()`) must be done before
330        """
331        method_names = self._last_used_methods
332
333        methods = (
334            self._feature_selection_method_objects
335            if not method_names
336            else [
337                method
338                for method in self._feature_selection_method_objects
339                if method.get_method_name() in method_names
340            ]
341        )
342        return dict(
343            zip(
344                [method.get_method_name() for method in methods],
345                [method.get_features_importance() for method in methods],
346            )
347        )

349    def get_stations_importance(self, target, method):
350        """
351        Generates a Stations importance for a target and a method. Feature selection (`select()`) must be done before
352
353        Args:
354            target (str) : target name (must be referenced in `target_columns` when `select()`)
355            method (str) : method name (must be referenced in `method_names` when `select()`, or None used)
356        """
357        stations_importance = self._stations_dataframe.copy()
358        stations_importance["nb_important_sensors"] = 0
359        stations_importance["max_importance_value"] = 0
360        stations_importance["sensors"] = ""
361        score = self.get_features_importance()[method]
362        for index in score.index:
363            x = re.search(self._stations_get_id_from_sensor_regex, index)
364            if x:
365                station_id = int(x.group(1))
366                importance_value = score[target][index]
367                stations_importance.loc[
368                    stations_importance[self._stations_id_column] == station_id,
369                    "nb_important_sensors",
370                ] += 1
371                stations_importance.loc[
372                    stations_importance[self._stations_id_column] == station_id,
373                    "sensors",
374                ] += f"{index} : {'{:.2f}'.format(importance_value)}\n</br>"
375                stations_importance.loc[
376                    (stations_importance[self._stations_id_column] == station_id)
377                    & (stations_importance["max_importance_value"] < importance_value),
378                    "max_importance_value",
379                ] = importance_value
380
381        return stations_importance
